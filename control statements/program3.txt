#include <stdio.h>

int main() {
    char sen[100];
    int sl;

    
    printf("Enter the sentence: ");
    fgets(sen, sizeof(sen), stdin);

    printf("Enter the screen length: ");
    scanf("%d", &sl);

 
    sen[sl] = '\0';

    int words = 0;
    for (int i = 0; sen[i] != '\0'; i++) {
        if (sen[i] == ' ') {
            words++;
        }
    }

    int space = sl - strlen(sen) + words;

    int sw = space / words;
    int extra = space % words;

    int wordCount = 0;
    for (int i = 0; sen[i] != '\0'; i++) {
        if (sen[i] != ' ') {
            printf("%c", sen[i]);
        } else {
            int star = sw;
            if (wordCount < extra) {
                star++;
            }
            for (int j = 0; j < star; j++) {
                printf("*");
            }
            wordCount++;
        }
    }

    printf("\n");

    return 0;
}
