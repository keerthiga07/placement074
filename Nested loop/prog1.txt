#include <stdio.h>

int main() {
    int mat[3][3];

  
    printf("Enter values for a 3x3 matrix:\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            printf("Enter value at position [%d][%d]: ", i + 1, j + 1);
            scanf("%d", &mat[i][j]);
        }
    }

  
    printf("\nMatrix:\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%d ", mat[i][j]);
        }
        printf("\n");
    }

    return 0;
}
